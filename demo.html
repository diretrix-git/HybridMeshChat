📋 Usage Example

// Initialize the system
const db = new DatabaseManager();
await db.initialize();

const transport = new TransportManager('device-123', db);

// Connect to room (tries WiFi first, falls back to Bluetooth)
await transport.connect(
  'room-abc',
  'http://192.168.1.100:3000',
  'password123'
);

// Send encrypted message
await transport.sendMessage('Hello mesh!', 'text');

// Automatic sync happens in background
// Messages are encrypted, deduplicated, and stored locally
